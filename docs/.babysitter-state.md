import { describe, it, expect, vi, beforeEach } from 'vitest';
import type { ObsidianConfig } from '../../types';

// ─── helpers reused across tests ────────────────────────────────────────────

function buildDOM() {
  document.body.innerHTML = `
    <fieldset id="obsidian-fieldset">
      <input id="obsidian-host" value="localhost" />
      <input id="obsidian-port" type="number" min="1" max="65535" value="27123" />
      <input id="obsidian-api-key" type="password" value="" />
      <button id="obsidian-test" type="button">Test</button>
      <span id="obsidian-status"></span>
      <input id="obsidian-profile-path" value="" />
      <button id="obsidian-browse-profile" type="button">Browse</button>
      <input id="obsidian-system-prompt-path" value="" />
      <button id="obsidian-browse-system-prompt" type="button">Browse</button>
      <datalist id="vault-files"></datalist>
    </fieldset>
    <div id="profile-form">
      <input data-profile="firstName" value="" />
      <input data-profile="email" value="" />
    </div>
  `;
}

const mockSendMessage = vi.fn();
vi.stubGlobal('chrome', { runtime: { sendMessage: mockSendMessage } });

vi.mock('../../lib/storage', () => ({
  getOllamaConfig: vi.fn(async () => ({ baseUrl: 'http://localhost:11434', model: 'llama3.2' })),
  getProfile: vi.fn(async () => ({})),
  setOllamaConfig: vi.fn(async () => undefined),
  setProfile: vi.fn(async () => undefined),
  getObsidianConfig: vi.fn(
    async (): Promise<ObsidianConfig> => ({
      host: 'localhost',
      port: 27123,
      apiKey: '',

  ),
  setObsidianConfig: vi.fn(async () => undefined),

import { getObsidianConfig, setObsidianConfig } from '../../lib/storage';

// ─── toggleProfileForm ───────────────────────────────────────────────────────

describe('toggleProfileForm', () => {
  beforeEach(() => {
    buildDOM();
  });

  it('hides #profile-form when profilePath is non-empty', async () => {
    const { toggleProfileForm } = await import('../main');
    toggleProfileForm({ host: 'localhost', port: 27123, apiKey: 'k', profilePath: 'Me.md' });
    expect(document.getElementById('profile-form')!.hidden).toBe(true);
  });

  it('shows #profile-form when profilePath is undefined', async () => {
    const { toggleProfileForm } = await import('../main');
    toggleProfileForm({ host: 'localhost', port: 27123, apiKey: 'k', profilePath: 'Me.md' });
    toggleProfileForm({ host: 'localhost', port: 27123, apiKey: 'k' });
    expect(document.getElementById('profile-form')!.hidden).toBe(false);
  });

  it('shows #profile-form when profilePath is an empty string', async () => {
    const { toggleProfileForm } = await import('../main');
    toggleProfileForm({ host: 'localhost', port: 27123, apiKey: 'k', profilePath: '' });
    expect(document.getElementById('profile-form')!.hidden).toBe(false);
  });
});

// ─── HTML structure (Task 3.1 acceptance criteria) ───────────────────────────

describe('Obsidian fieldset structure', () => {

describe('load() — Obsidian fields', () => {

// ─── save() — Obsidian config ────────────────────────────────────────────────

describe('save() — Obsidian config persistence', () => {
  beforeEach(() => {
    buildDOM();
    vi.mocked(getObsidianConfig).mockResolvedValue({ host: 'localhost', port: 27123, apiKey: '' });
    vi.mocked(setObsidianConfig).mockResolvedValue(undefined);
  });

  it('calls setObsidianConfig with values from inputs', async () => {
    (document.getElementById('obsidian-host') as HTMLInputElement).value = 'myhost';
    (document.getElementById('obsidian-port') as HTMLInputElement).value = '9000';
    (document.getElementById('obsidian-api-key') as HTMLInputElement).value = 'secret';
    (document.getElementById('obsidian-profile-path') as HTMLInputElement).value = 'Me.md';

    const { save } = await import('../main');
    await save();

    expect(setObsidianConfig).toHaveBeenCalledWith(
      expect.objectContaining({
        host: 'myhost',
        port: 9000,
        apiKey: 'secret',
        profilePath: 'Me.md',
      }),
    );
  });

  it('does not collect [data-profile] inputs when profilePath is non-empty', async () => {
    (document.getElementById('obsidian-profile-path') as HTMLInputElement).value = 'Me.md';
    (document.querySelector('[data-profile="firstName"]') as HTMLInputElement).value = 'Alice';

    const { save, setProfileSpy } = await import('../main');
    await save();

    expect(setProfileSpy).not.toHaveBeenCalled();
  });

  it('collects [data-profile] inputs when profilePath is empty', async () => {
    (document.getElementById('obsidian-profile-path') as HTMLInputElement).value = '';
    (document.querySelector('[data-profile="firstName"]') as HTMLInputElement).value = 'Alice';

    const { save, setProfileSpy } = await import('../main');
    await save();

    expect(setProfileSpy).toHaveBeenCalled();
  });
});

// ─── Test button ─────────────────────────────────────────────────────────────

describe('Test button (OBSIDIAN_TEST_CONNECTION)', () => {
  beforeEach(() => {
    buildDOM();
    vi.mocked(getObsidianConfig).mockResolvedValue({
      host: 'localhost',
      port: 27123,
      apiKey: 'key',
    });
  });

  it('sends OBSIDIAN_TEST_CONNECTION when clicked', async () => {
    mockSendMessage.mockResolvedValue({ ok: true });
    const { wireTestButton } = await import('../main');
    wireTestButton();
    document.getElementById('obsidian-test')!.click();
    await Promise.resolve();
    expect(mockSendMessage).toHaveBeenCalledWith({ type: 'OBSIDIAN_TEST_CONNECTION' });
  });

  it('updates #obsidian-status with success text on ok response', async () => {
    mockSendMessage.mockResolvedValue({ ok: true });
    const { wireTestButton } = await import('../main');
    wireTestButton();
    document.getElementById('obsidian-test')!.click();
    await new Promise((r) => setTimeout(r, 10));
    expect(document.getElementById('obsidian-status')!.textContent).toMatch(/connected/i);
  });

  it('updates #obsidian-status with error text on failure', async () => {
    mockSendMessage.mockResolvedValue({ ok: false, error: 'timeout' });
    const { wireTestButton } = await import('../main');
    wireTestButton();
    document.getElementById('obsidian-test')!.click();
    await new Promise((r) => setTimeout(r, 10));
    expect(document.getElementById('obsidian-status')!.textContent).toMatch(/timeout/i);
  });

  it('disables the test button during the in-flight request', async () => {
    let resolve!: () => void;
    mockSendMessage.mockReturnValue(
      new Promise<{ ok: true }>((r) => (resolve = () => r({ ok: true }))),
    );
    const { wireTestButton } = await import('../main');
    wireTestButton();
    const btn = document.getElementById('obsidian-test') as HTMLButtonElement;
    btn.click();
    expect(btn.disabled).toBe(true);
    resolve();
    await new Promise((r) => setTimeout(r, 10));
    expect(btn.disabled).toBe(false);
  });
});

// ─── Browse buttons ───────────────────────────────────────────────────────────

describe('Browse buttons (OBSIDIAN_LIST_FILES)', () => {
  beforeEach(() => {
    buildDOM();
    vi.mocked(getObsidianConfig).mockResolvedValue({
      host: 'localhost',
      port: 27123,
      apiKey: 'key',
    });
  });

  it('are disabled when apiKey input is empty', async () => {
    (document.getElementById('obsidian-api-key') as HTMLInputElement).value = '';
    const { syncBrowseButtonState } = await import('../main');
    syncBrowseButtonState();
    expect((document.getElementById('obsidian-browse-profile') as HTMLButtonElement).disabled).toBe(
      true,
    );
    expect(
      (document.getElementById('obsidian-browse-system-prompt') as HTMLButtonElement).disabled,
    ).toBe(true);
  });

  it('are enabled when apiKey input is non-empty', async () => {
    (document.getElementById('obsidian-api-key') as HTMLInputElement).value = 'secret';
    const { syncBrowseButtonState } = await import('../main');
    syncBrowseButtonState();
    expect((document.getElementById('obsidian-browse-profile') as HTMLButtonElement).disabled).toBe(
      false,
    );
  });

  it('sends OBSIDIAN_LIST_FILES when profile browse is clicked', async () => {
    mockSendMessage.mockResolvedValue({ ok: true, files: ['a.md', 'b.md'] });
    const { wireBrowseButtons } = await import('../main');
    wireBrowseButtons();
    document.getElementById('obsidian-browse-profile')!.click();
    await new Promise((r) => setTimeout(r, 10));
    expect(mockSendMessage).toHaveBeenCalledWith({ type: 'OBSIDIAN_LIST_FILES' });
  });

  it('populates #vault-files datalist with returned .md paths', async () => {
    mockSendMessage.mockResolvedValue({ ok: true, files: ['Notes/A.md', 'Profile/Me.md'] });
    const { wireBrowseButtons } = await import('../main');
    wireBrowseButtons();
    document.getElementById('obsidian-browse-profile')!.click();
    await new Promise((r) => setTimeout(r, 20));
    const dl = document.getElementById('vault-files') as HTMLDataListElement;
    const values = Array.from(dl.options).map((o) => o.value);
    expect(values).toContain('Notes/A.md');
    expect(values).toContain('Profile/Me.md');
  });
});
